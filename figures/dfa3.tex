    },
    accepting/.style={
        double,
        double distance=2pt
    },
    initial/.style={
        initial by arrow,
        initial distance=7pt,
        initial text={}
    }
}

  (q1) edge[bend left, above] node[above] {$b$} (q0);

  (q2) edge[loop above] node {$a,b$} ();
