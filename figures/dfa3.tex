\usepackage{tikz}

\tikzset{
    state/.style={
        circle,

        minimum size=20pt,
        inner sep=0pt
    },
    accepting/.style={
        double,
        double distance=2pt
    },
    initial/.style={
        initial by arrow,
        initial distance=7pt,
        initial text={}
    }
}

  (q1) edge[bend left=40] node[above] {$\mathtt{a}$} (q2)
  (q1) edge[out=90, in=90, looseness=2] node[above] {$\mathtt{b}$} (q0);

  (q2) edge[loop above] node {$\mathtt{a,b}$} ();
